\section{Производная. Максимумы и минимумы.}

\resetz

{\boldmath
\textbf{1) } \textbf{Таблица производных простейших функций}:
$$\begin{array}{|c|c|c|c|c|c|c|c|c|c|c|c|}
\hline f(x) & C &x^a & \sqrt{x} &\sin x &\cos x &\tg x &\ctg x &e^x &a^x  &\log_ax  &\ln x  \\
\hline
 f'(x) & 0 &ax^{a-1}& \dfrac{1}{2\sqrt{x}} &\cos x &-\sin x &1+\tg^2x &-1-\ctg^2x &e^x  &a^x\ln a &\dfrac{1}{x\ln a} & \dfrac{1}{x} \\
\hline
\end{array}
$$
\textbf{2)} \textbf{Правила дифференцирования}:
$$( (f(x)\pm g(x))'=f'(x)\pm g'(x);\quad \left(f(x)\cdot g(x) \right)'=f'(x)g(x)+f(x)g'(x);\quad$$
$$ \left(\dfrac{f(x)}{g(x)} \right)'=\dfrac{f'(x)g(x)-f(x)g'(x)}{g^2(x)},\;g(x)\ne0;\quad (f(g(x)))'=f'(g(x))\cdot g'(x). $$

\textbf{\textit{ Если во всех точках некоторого интервала
производная функции $f(x)$ больше (соответственно меньше) нуля, то
функция $f(x)$ возрастает (соответственно убывает) на этом
интервале. }}}
