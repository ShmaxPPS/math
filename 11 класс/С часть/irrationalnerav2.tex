[\section*{Иррациональные неравенства.}
\setcounter{z}{0}
{
 \it
$$
 1) \sqrt{f(x)}<g(x) \Longleftrightarrow \left\lbrace \begin{array}{l}
0\leq f(x)<g^2(x)\\
g(x)\geq0
\end{array}
\right.;
 $$
 $$
 2) \sqrt{f(x)}>g(x) \Longleftrightarrow \left[ \begin{array}{l}
{\left\lbrace \begin{array}{l}
f(x)>g^2(x)\\
g(x)\geq0
\end{array}
\right.} \\
{\left\lbrace \begin{array}{l}
f(x)\geq0\\
g(x)<0
\end{array}\right.}
\end{array}\right.
 $$
}
]


\textbf{Решите неравенства:}


\zad   $\sqrt{x-8}>3.$

\zad   $\sqrt{2-4x}<4.$

\hans{ (-3,5;0,5].}

\zad   $\sqrt{3x-6}<3.$

\zad   $\sqrt{2-x}<-x.$

\zad   $\sqrt{x+1}>x-1.$

\hans{  [-1;3).}

\zad   $\sqrt{2x-1}<x.$

\zad   $\sqrt{x^2+3x+2}>3-x.$

\zad   $\sqrt{x+9}>3-x.$


\hans{  (0;+\infty).}


\zad   $x-3\geq\sqrt{9-x^2}.$

\hans{  \{3\}.}


\zad $ \sqrt{3+4x}\ge \sqrt{6x-9}.$


\zad $ \sqrt{22+x}-\sqrt{3-x}<5.$


\zad   $\sqrt{x+2}+\sqrt{3-x}<3.$

\hans{  [-2;-1)\cup(2;3].}


\zad   $\sqrt{x-1}+\sqrt{x+2}\leq1.$


\zad   $\sqrt{2-\sqrt{x+3}}<\sqrt{x+4}.$



\zad   $\sqrt{2x+\sqrt{6x^2+1}}\leq x+1.$


\hans{  \{0\}\cup[2;+\infty).}


\zad   $\sqrt{\dfrac{3x-1}{2-x}}>1.$

\hans{  (0,75;2).}



\zad   $(9-x^2)\sqrt{x+4}\geq0.$

\hans{  \{-4\}\cup[-3;3].}

\zad   $(x+10)\sqrt{x-4}\leq0.$

\hans{  \{4\}.}

\zad   $(x-12)\sqrt{x-3}\leq0.$

\zad   $(x-3)\sqrt{x^2+x-2}\geq0.$
