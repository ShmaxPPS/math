
\section*{Логарифмические неравенства.}
\setcounter{z}{0}

\begin{normalsize} \textbf{\textit{Свойства логарифмической функции}}\quad
$y=\log_ax$ \; ${(a>0,\; a\neq1)}:$
\begin{enumerate}
\item Область определения функции --- множество всех положительных
действительных чисел ($x>0$). \item Область значений функции ---
множество всех действительных чисел. \item  При $a>1$ функция
возрастает:\quad $0<x_1<x_2\Longleftrightarrow
\log_a{x_1}<\log_a{x_2}$. \item  При $0<a<1$ функция убывает:
\quad $0<x_1<x_2\Longleftrightarrow {\log_a{x_1}>\log_a{x_2}}$.
% \item $\log_a{x_1}=\log_a{x_2}\Longleftrightarrow x_1=x_2>0.$
% \item $\log_a{a^x}=x$ для всех $x\in\R$, \quad $a^{\log_ax}=x$ для
% всех $x>0.$
\end{enumerate}
\end{normalsize}
