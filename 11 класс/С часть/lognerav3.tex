
\section{Логарифмические неравенства.}
\setcounter{z}{0}








\zad $\log_2(x^2+3x)\le 2.$

\zad $\log_{1/3}\left(\dfrac{2-3x}{x} \right)\ge -1.$

\zad $\log_3(x^2-x)\le \log_3(3x+2).$

\zad $\log_5 x+\log_{25}x<\log_{1/5}\sqrt 3.$

\zad $\log^2_2(5-x)-6\log_2(5-x)+9\le 0.$

\zad $\dfrac4{\lg 10x}-\dfrac5{\lg100x}\ge 0.$

\zad $\dfrac{1}{1-\lg x}< \dfrac{2\lg x-5}{1+\lg x}.$

\zad $\log_2(\log_{1/3}(\log_5x))>0.$


\zad $\sqrt{1-\log_{1/2}x} <2.$


\zad $\dfrac{\sqrt {2x+1}}{2+\log_{1/2}(x+1)}\ge 0.$

\zad $\left(\dfrac12\right)^{\log_{1/9}(2x^2-3x+1)}>1.$


\zad $\log_x(1-2x)<1.$

\zad $\log_{2x}\log_3\dfrac{x+1}{x-1}<
\log_{\frac1{2x}}\log_{1/3}\dfrac{x-1}{x+1}.$

\zad $\log_{|x|}\log_2(4^x-12)\le 1.$

\zad $\log_{\log_2(x/2)}(x^2-10x+22)>0.$

\section*{Задачи для самостоятельного решения.}
\setcounter{z}{0}


\zad $\log_{1/\sqrt 5}(-x^2+6x+3)\ge -2.$

\zad $\log_2\left(\dfrac{x+1}{x} \right)>0.$

\zad $\log_3(x+2)+\log_3(x-4)\le 0.$

\zad $\lg (x-2)+\lg(x-5)\le \lg 4.$

\zad $\dfrac1{\log_2 x-4}>\dfrac1{\log_2x}.$

\zad $2+\dfrac{\log_2^2x}{1+\log_2x}>\log_2x.$

\zad $\dfrac1{5-\lg x}+\dfrac2{1+\lg x}<1.$

\zad $(4x-1)\log_2x\ge 0.$

\zad $\left(\dfrac12\right)^{\log_3\log_{1/5}(x^2-4/5)}>1.$

\zad $\log_x(\log_2(4^x-12))\le 1.$
