\textbf{\emph{Это должен знать каждый:}}

\begin{itemize}
    \item $|A|<B \Longleftrightarrow -B<A<B;$
    \item $|A|>B \Longleftrightarrow \left[ \begin{array}{l}
  A>B \\
  A<-B \\
\end{array}  \right.$
 \item $|A|<|B| \Longleftrightarrow A^2<B^2;$
\end{itemize}
